                    updateThemeButtonText(true);
                } else {
                    document.body.classList.remove('light-theme');
                    document.body.classList.add('dark-theme');
                    updateThemeButtonText(false);
                }

                    document.body.classList.add('light-theme');
                    localStorage.setItem('cognos_theme', 'light');
                    updateThemeButtonText(true);
                } else {

            function checkExistingSession() {
                const isLoggedIn = localStorage.getItem('cognos_auth');

                    displayUserNameSpan.innerText = savedUser;
                    loginOverlay.classList.add('hidden');
                    dashboardApp.style.display = 'flex';
                    buildDashboardGrid();

            function handleLogin() {
                const username = usernameInput.value;
                const password = passwordInput.value;
                if (username && username.trim() !== "" && password && password.trim() !== "") {
                    localStorage.setItem('cognos_auth', 'true');
                    localStorage.setItem('cognos_user', username.trim());
                    displayUserNameSpan.innerText = username.trim();
                    loginOverlay.classList.add('hidden');
                    dashboardApp.style.display = 'flex';
                    buildDashboardGrid();
